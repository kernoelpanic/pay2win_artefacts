def fig_dist(x_max=300):
    x = np.arange(0, x_max)
    fig, ax = plt.subplots(figsize=(16, 9))

    plt.plot(x, [scipy.stats.poisson.pmf(i,(60*24)/10) for i in range(0,x_max)] )
    ax.axvline(144-12) # - std deviation
    ax.axvline(144+12) # + std deviation
    
    # tidy up the figure
    ax.grid(True)
    ax.set_title("Distribution") 
    ax.set_xlabel("x")
    ax.set_ylabel("P(x)")
    ax.set_xlim(80,200)
    ax.set_ylim(0,0.04)

    plt.show()
